1function obj=objSR(X,W,A)
 2% compute the objective function for sphere approximation
 3% Input:
 4%     X - input square root density data (d-by-n)
 5%     W - coefficient matrix (m-by-n)
 6%     A - atoms in dictionary (d-by-m)
 7% Output:
 8%     obj - the value of objective function
 9
10% initialization
11d = size(X,1);
12n = size(W,2);
13
14T = zeros(d,n);
15parfor i = 1:n
16    LOG = logmapSRi(X(:,i),A); % LOG d-by-m matrix with the jth column log(x_i,a_j)
17    T(:,i) = LOG*W(:,i); % T: d-by-n matrix with the ith column sum of W_ji*log(x_i,a_j)
18end
19obj = trace(T'*T);
