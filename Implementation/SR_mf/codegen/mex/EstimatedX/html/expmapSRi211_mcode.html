1function Y = expmapSRi(X,V)
 2%#codegen
 3% compute the exp map on sphere exp_X(V)
 4% Input:
 5%     X - d-by-m matrix, with each column a squre root density x
 6%     V - d-by-m matrix, with each column a tangent vector v
 7%     to ensure exp map is a bijection, |v| \in [0,\pi]
 8% Output:
 9%     Y - expmap (d-by-m matrix), each column exp_x(v)
10
11[d,~] = size(X);
12normV = sqrt(sum(V.^2,1)); %1-by-m
13normV0 = normV;
14normV(normV == 0) = 1;
15V = V./(repmat(normV,d,1));
16Y = X*diag(cos(normV0))+V*diag(sin(normV0));
