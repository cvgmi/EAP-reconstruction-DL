1function gradA = gradientASR(X,W,A)
 2%#codegen
 3% compute the gradient with respect to A
 4% Input:
 5%     X - input square root density data (d-by-n)
 6%     W - coefficient matrix (m-by-n)
 7%     A - initial atoms in dictionary (d-by-m)
 8% Output:
 9%     gradA - the gradient with respect to A (d-by-m)
10
11%% Initialization
12d = size(X,1);
13[m,n] = size(W);
14Wt = W';
15
16gradA = zeros([d,m]);
17
18%% Main loop for each atom Aj
19parfor j=1:m
20    for i=1:n
21%         for k=1:m
22%             Lg(:,k)=logmapSR(X(:,i),A(:,k));
23%         end
24        Lg = logmapSRi(X(:,i),A);
25        M = Lg*W(:,i);
26
27        uij = A(:,j)-(X(:,i)'*A(:,j))*X(:,i);
28        normu = norm(uij);
29        inner = X(:,i)'*A(:,j);
30        if (normu ~= 0)&&(inner^2 < 1-1E-5)
31            dA = (eye(d)/normu-uij*uij'/(normu^3))*(eye(d)-X(:,i)*X(:,i)');
32            dA = dA*acos(inner)-uij*X(:,i)'/normu/sqrt(1-inner^2);
33            gradA(:,j) = gradA(:,j) + Wt(i,j)*dA'*M;
34        end
35    end
36end
37gradA = 2*gradA;
