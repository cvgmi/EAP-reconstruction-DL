1function A = optASR_sigma_disp(X,W,A0,param)
 2%#codegen
 3% minimize the square root density cost function with respect to A
 4% Input:
 5%     X - input square root density data (d-by-n)
 6%     W - coefficient matrix (m-by-n)
 7%     A0 - initial atoms in dictionary (d-by-m)
 8% Output:
 9%     A - final atoms in dictionary (d-by-m)
10
11coder.extrinsic('fprintf'); 
12%% Initialization
13n = size(W,2);
14A = A0;
15s = 0;
16
17% parameters
18% maxiter = 5;
19% eps = 1e-7;
20% alpha = 1;
21% beta = 0.1;
22% sigma = 0.9;
23% maxk = 30;
24
25% parameters
26maxiterls = param.maxiterls;
27epsls = param.epsls;
28alpha = param.alpha;
29beta = param.beta;
30sigma = param.sigma;
31maxk = param.maxk;
32
33% initial objective values
34obj_old = objSR(X,W,A0);
35obj_new = 2*obj_old;
36
37%% Main loop
38for iter=1:maxiterls
39    % compute gradient
40    gradA = gradientASR(X,W,A0);
41    % project to the tangent plane
42    eta = projectT(A0,gradA);
43    % set the descent direction (negative gradient)
44    eta = -eta;
45    % update the dictionary atoms
46    for k=1:maxk
47        A = expmapSRi(A0,alpha*beta^k*eta);
48        temp = diag(eta'*eta);
49        temp = sqrt(temp);
50        s = sigma*alpha*beta^k*sum(temp);
51        
52        obj_new = objSR(X,W,A);
53        if(obj_new < obj_old - s)
54            break;
55        end
56    end
57    
58    fprintf('maxk: %d\n',k);
59    if obj_old-obj_new < s
60        fprintf('maxk not big enough\n');
61    end
62
63    if((obj_old-obj_new)/n<epsls)
64        fprintf('eps reached\n');
65        break;
66    end
67    A0 = A;
68    obj_old = obj_new;
69end
70fprintf('maxiter: %d\n', iter);
71
72
