1function P = projectTP(AA,GG)
 2%#codegen
 3% project the gradient G to the tangent plane at A
 4% Input:
 5%     AA - atoms in dictionary (d-by-N-by-m)
 6%     GG - gradient with respect to A (d-by-N-by-m)
 7% Output:
 8%     P - projected gradient (d-by-N-by-m)
 9
10[d,N,m] = size(AA);
11P = zeros([d,N,m]);
12
13parfor i = 1:N
14    A = squeeze(AA(:,i,:));
15    G = squeeze(GG(:,i,:));
16    P(:,i,:) = projectT(A,G);
17end
