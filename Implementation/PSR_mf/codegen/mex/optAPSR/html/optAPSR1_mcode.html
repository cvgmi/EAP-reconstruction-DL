1function A = optAPSR(X,W,A0,param)
 2%#codegen
 3% assert(isa(X,'double'));
 4% assert(isa(W,'double'));
 5% assert(isa(A0,'double'));
 6% assert(all(size(X) == [64 121]));
 7% assert(all(size(W) == [100 121]));
 8% assert(all(size(A0) == [64 100]));
 9% minimize the square root density cost function with respect to A
10% Input:
11%     X - input square root density data (d-by-N-by-n)
12%     W - coefficient matrix (m-by-n)
13%     A0 - initial atoms in dictionary (d-by-N-by-m)
14% Output:
15%     A - final atoms in dictionary (d-by-N-by-m)
16
17coder.extrinsic('fprintf');
18%% Initialization
19[m,n] = size(W);
20A = A0;
21
22% parameters
23maxiter = param.maxiterAPSR;
24eps = param.eps;
25alpha = param.alpha;
26beta = param.beta;
27%sigma = param.sigma;
28maxk = param.maxk;
29
30% initial objective values
31obj_old = objPSR(X,W,A0);
32obj_new = 2*obj_old;
33
34%% Main loop
35for iter=1:maxiter
36    % compute gradient
37    gradA = gradientAPSR(X,W,A0);
38    % project to the tangent plane
39    eta = projectTP(A0,gradA);
40    % set the descent direction (negative gradient)
41    eta = -eta;
42    % update the dictionary atoms
43    for k=1:maxk
44        %sum = 0;
45        parfor i =1:m
46        %for i=1:m
47            A(:,:,i) = expmapPSR(A0(:,:,i),alpha*beta^k*eta(:,:,i));
48            %sum = sum + sigma*alpha*beta^k*sqrt(eta(:,i)'*eta(:,i));
49        end
50        obj_new = objPSR(X,W,A);
51        if(obj_new < obj_old)
52            break;
53        end
54    end
55    fprintf('maxk: %d\n',k);
56    if obj_old<obj_new
57        fprintf('maxk not big enough\n');
58    end
59
60    if((obj_old-obj_new)/n<eps)
61        fprintf('eps reached\n');
62        break;
63    end
64    A0 = A;
65    obj_old = obj_new;
66end
67fprintf('maxiter: %d\n', iter);
68
