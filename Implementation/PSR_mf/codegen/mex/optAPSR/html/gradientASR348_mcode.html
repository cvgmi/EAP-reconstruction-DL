1function gradA = gradientASR(X,W,A)
 2%#codegen
 3% compute the gradient with respect to A
 4% Input:
 5%     X - input square root density data (d-by-n)
 6%     W - coefficient matrix (m-by-n)
 7%     A - initial atoms in dictionary (d-by-m)
 8% Output:
 9%     gradA - the gradient with respect to A (d-by-m)
10
11%% Initialization
12d = size(X,1);
13[m,n] = size(W);
14Wt = W';
15
16gradA = zeros([d,m]);
17% Lg = zeros(d,m);
18%% Main loop for each atom Aj
19parfor j=1:m
20    for i=1:n
21        Lg = logmapSRi(X(:,i),A);
22        M = Lg*W(:,i);
23
24        uij = A(:,j)-(X(:,i)'*A(:,j))*X(:,i);
25        normu = norm(uij);
26        inner = X(:,i)'*A(:,j);
27        if (normu ~= 0)&&(inner^2 < 1-1E-5)
28            dA = (eye(d)/normu-uij*uij'/(normu^3))*(eye(d)-X(:,i)*X(:,i)');
29            dA = dA*acos(inner)-uij*X(:,i)'/normu/sqrt(1-inner^2);
30            gradA(:,j) = gradA(:,j) + Wt(i,j)*dA'*M;
31        end
32    end
33end
34gradA = 2*gradA;
