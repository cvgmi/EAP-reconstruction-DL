1function P = projectT(A,G)
 2%#codegen
 3% project the gradient G to the tangent plane at A
 4% Input:
 5%     A - atoms in dictionary (d-by-m)
 6%     G - gradient with respect to A (d-by-m)
 7% Output:
 8%     P - projected gradient (d-by-m)
 9
10[d,m] = size(A);
11P = zeros([d,m]);
12
13%for i=1:m
14parfor i=1:m
15    P(:,i) = (eye(d)-A(:,i)*A(:,i)')*G(:,i);
16end
