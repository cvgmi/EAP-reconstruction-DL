1function [A, obj_new] = optAPSR_sigma(X,W,A0,param,obj_old)
 2%#codegen
 3% minimize the square root density cost function with respect to A
 4% Input:
 5%     X - input square root density data (d-by-N-by-n)
 6%     W - coefficient matrix (m-by-n)
 7%     A0 - initial atoms in dictionary (d-by-N-by-m)
 8% Output:
 9%     A - final atoms in dictionary (d-by-N-by-m)
10
11coder.extrinsic('fprintf'); 
12%% Initialization
13[m,n] = size(W);
14A = A0;
15
16% parameters
17maxiter = param.maxiterAPSR;
18eps = param.eps;
19alpha = param.alpha;
20beta = param.beta;
21sigma = param.sigma;
22maxk = param.maxk;
23
24% initial objective values
25obj_new = 2*obj_old;
26
27%% Main loop
28for iter=1:maxiter
29    % compute gradient
30    gradA = gradientAPSR(X,W,A0);
31    % project to the tangent plane
32    eta = projectTP(A0,gradA);
33    % set the descent direction (negative gradient)
34    eta = -eta;
35    s = 0;
36    % update the dictionary atoms
37    for k=1:maxk
38        s = 0;
39        for i=1:m
40            A(:,:,i) = expmapPSR(A0(:,:,i),alpha*beta^k*eta(:,:,i));
41            temp = sqrt(sum(sum(squeeze(eta(:,:,i)).^2)));
42            s = s + sigma*alpha*beta^k*temp;
43        end
44        obj_new = objPSR(X,W,A);
45        if(obj_new < obj_old - s)
46            break;
47        end
48    end
49    fprintf('---maxk: %d\n',k);
50    if obj_old-obj_new<s
51        fprintf('---**maxk not big enough**\n');
52    end
53
54    if((obj_old-obj_new)/n<eps)
55        fprintf('---**eps reached**\n');
56        break;
57    end
58    A0 = A;
59    obj_old = obj_new;
60end
61fprintf('---maxiter: %d\n', iter);
62
