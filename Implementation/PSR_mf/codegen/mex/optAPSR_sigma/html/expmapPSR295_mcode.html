1function Y = expmapPSR(X,V)
 2%#codegen
 3% compute the exp map on product space of sphere exp_X(V)
 4% Input:
 5%     X, V - d-by-N matrices
 6%     to ensure exp map is a bijection, |v| \in [0,\pi]
 7% Output:
 8%     Y - expmap (d-by-N vector)
 9
10% same as expmapSRi in DL_SR
11[d,~] = size(X);
12normV = sqrt(sum(V.^2,1)); %1-by-m
13normV0 = normV;
14normV(normV == 0) = 1;
15V = V./(repmat(normV,d,1));
16Y = X*diag(cos(normV0))+V*diag(sin(normV0));
