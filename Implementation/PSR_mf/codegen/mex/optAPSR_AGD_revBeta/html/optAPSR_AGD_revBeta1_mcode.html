1function [A, obj_new, param] = optAPSR_AGD_revBeta(X,W,A0,param,obj_old)
 2%#codegen
 3% minimize the square root density cost function with respect to A
 4% Input:
 5%     X - input square root density data (d-by-N-by-n)
 6%     W - coefficient matrix (m-by-n)
 7%     A0 - initial atoms in dictionary (d-by-N-by-m)
 8% Output:
 9%     A - final atoms in dictionary (d-by-N-by-m)
10
11coder.extrinsic('fprintf'); 
12%% Initialization
13[m,n] = size(W);
14
15% initialization for AGD
16A = A0;
17B0 = A0;
18B = B0;
19lambda0 = 0;
20lambda = (1+sqrt(1+4*lambda0^2))/2;
21lambda0 = lambda;
22
23
24% parameters
25maxiterAGD = param.maxiterAGD;
26epsAGD = param.epsAGD;
27beta = param.beta;
28
29% initial objective values
30% obj_old = objPSR(X,W,A0);
31obj_new = 2*obj_old;
32
33s = 1;
34
35%% Main loop
36while s < maxiterAGD
37    
38    lambda = (1+sqrt(1+4*lambda0^2))/2;
39    gamma = (1-lambda0)/lambda;
40    
41    % compute gradient
42    gradA = gradientAPSR(X,W,A0);    
43    % project to the tangent plane
44    eta = projectTP(A0,gradA);
45    % set the descent direction (negative gradient)
46    eta = -eta;
47    % update the dictionary atoms
48
49    V = zeros(size(B));
50    for i = 1:m
51        B(:,:,i) = expmapPSR(A0(:,:,i),1/beta*eta(:,:,i));
52        V(:,:,i) = logmapPSR(B(:,:,i),B0(:,:,i));
53        A(:,:,i) = expmapPSR(B(:,:,i),gamma*V(:,:,i));
54    end
55    obj_new = objPSR(X,W,A);
56    
57    if obj_old<obj_new % if obj increases, reverse and increase beta
58        beta = beta*2;
59        A = A0;
60        B = B0;
61        lambda = lambda0;
62        obj_new = obj_old;
63        s = s-1;
64        fprintf(1,'obj increased, reverse and increase beta\n');
65    elseif((obj_old-obj_new)/n<epsAGD)
66        fprintf(1,'eps reached\n');
67        break;
68    end
69   
70    A0 = A;
71    B0 = B;
72    lambda0 = lambda;
73    obj_old = obj_new;
74    s = s+1;
75end
76param.beta = beta;
77fprintf('maxiter: %d\n', s);
