1function T = costMatrixPSR(X,AA)
 2% compute the matrix T for the optimization step wrt W
 3% Input:
 4%     x - one square root density field data (d-by-N)
 5%     A - atoms in dictionary (d-by-N-by-m)
 6% Output:
 7%     T - symmetric matrix (m-by-m)
 8
 9[~,N,m] = size(AA);
10T = zeros(m,m);
11parfor j = 1:N
12    x = squeeze(X(:,j));
13    A = squeeze(AA(:,j,:));
14    V = logmapSRi(x,A);
15    T = T+V'*V;
16end
