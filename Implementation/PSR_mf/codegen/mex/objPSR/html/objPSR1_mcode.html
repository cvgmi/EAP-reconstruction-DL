1function obj=objPSR(X,W,A)
 2%#codegen
 3% assert(isa(X,'double'));
 4% assert(isa(W,'double'));
 5% assert(isa(A,'double'));
 6% assert(all(size(X) == [64 121]));
 7% assert(all(size(W) == [100 121]));
 8% assert(all(size(A) == [64 100]));
 9
10% compute the objective function for product of sphere approximation
11% Input:
12%     X - input square root density field data (d-by-N-by-n)
13%     W - coefficient matrix (m-by-n)
14%     A - atoms in dictionary (d-by-N-by-m)
15% Output:
16%     obj - the value of objective function
17
18% initialization
19[d,N,n] = size(X);
20[m,~] = size(W);
21
22% main
23obj = 0;
24parfor i=1:n
25    LOG = logmapPSRi(X(:,:,i),A);
26    LOG = reshape(LOG,[d*N m]);
27    M = reshape(LOG*W(:,i),[d N]);
28    obj = obj + trace(M'*M);
29end
