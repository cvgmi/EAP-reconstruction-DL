1function V = logmapSRi(x,Y)
 2%#codegen
 3% compute the log map on sphere log_x(Y) 
 4% Input:
 5%     x - d-by-1 vector
 6%     Y - d-by-m matrix, each column an atom (d-by-1)
 7% Output:
 8%     V - logmap of each column of Y at x (d-by-m matrix)
 9
10[d,m] = size(Y);
11temp = x'*Y;  %1-by-m
12U = Y - repmat(x,1,m)*diag(temp); % d-ny-m
13normU = sqrt(sum(U.^2,1)); %1-by-m
14normU(normU == 0) = 1;
15U = U./(repmat(normU,d,1));
16temp(temp>1) = 1;
17temp(temp<-1) = -1;
18V = U*diag(acos(temp)); 
19
