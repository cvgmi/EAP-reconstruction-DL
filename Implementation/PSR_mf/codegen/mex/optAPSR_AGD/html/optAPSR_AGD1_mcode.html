1function [A, obj_new, param] = optAPSR_AGD(X,W,A0,param,obj_old)
 2%#codegen
 3% minimize the square root density cost function with respect to A
 4% Input:
 5%     X - input square root density data (d-by-N-by-n)
 6%     W - coefficient matrix (m-by-n)
 7%     A0 - initial atoms in dictionary (d-by-N-by-m)
 8% Output:
 9%     A - final atoms in dictionary (d-by-N-by-m)
10
11coder.extrinsic('fprintf'); 
12%% Initialization
13[m,n] = size(W);
14
15% initialization for AGD
16A = A0;
17B0 = A0;
18B = B0;
19lambda0 = 0;
20lambda = (1+sqrt(1+4*lambda0^2))/2;
21lambda0 = lambda;
22
23
24% parameters
25maxiterAGD = param.maxiterAGD;
26epsAGD = param.epsAGD;
27beta = param.beta;
28
29% initial objective values
30% obj_old = objPSR(X,W,A0);
31obj_new = 2*obj_old;
32
33%% Main loop
34for s=1:maxiterAGD
35    
36    lambda = (1+sqrt(1+4*lambda0^2))/2;
37    gamma = (1-lambda0)/lambda;
38    
39    % compute gradient
40    gradA = gradientAPSR(X,W,A0);    
41    % project to the tangent plane
42    eta = projectTP(A0,gradA);
43    % set the descent direction (negative gradient)
44    eta = -eta;
45    % update the dictionary atoms
46
47    V = zeros(size(B));
48    for i = 1:m
49        B(:,:,i) = expmapPSR(A0(:,:,i),1/beta*eta(:,:,i));
50        V(:,:,i) = logmapPSR(B(:,:,i),B0(:,:,i));
51        A(:,:,i) = expmapPSR(B(:,:,i),gamma*V(:,:,i));
52    end
53    obj_new = objPSR(X,W,A);
54    
55    if((obj_old-obj_new)/n<epsAGD)
56        fprintf('eps reached\n');
57        break;
58    end
59   
60    A0 = A;
61    B0 = B;
62    lambda0 = lambda;
63    obj_old = obj_new;
64end
65fprintf('maxiter: %d\n', s);
