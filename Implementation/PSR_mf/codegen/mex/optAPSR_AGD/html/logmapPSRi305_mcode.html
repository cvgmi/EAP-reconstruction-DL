1function VV = logmapPSRi(X,YY)
 2%#codegen
 3% compute the log map on the product space of spheres log_X(Y)
 4% Input:
 5%     X - d-by-N matrices, a point in the product space
 6%     YY - d-by-N-by-m matrices, m points in the product space 
 7% Output:
 8%     V - logmap of each d-by-N matrix in Y (d-by-N-m matrix)
 9
10[~,N,~] = size(YY);
11VV = zeros(size(YY));
12% do logmap for each column of X (d-by-1 vector) and the corresponding
13% columns in YY
14parfor i = 1:N
15    x = squeeze(X(:,i));
16    Y = squeeze(YY(:,i,:));
17    VV(:,i,:) = logmapSRi(x,Y);
18end
